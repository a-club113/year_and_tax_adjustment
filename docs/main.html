194    def handle_drop(self, event):
195        """
196            handle drop event
197        """
198        filenames = re.findall(r'{([^}]*)}', event.data)
199
200        if not filenames:
201            filenames = event.data.split()
202
203        pdf_files = [f.strip('\'\"') for f in filenames if f.lower().strip('\'\"').endswith('.pdf')]
204
205        if pdf_files:
206            self.load_pdf(pdf_files[0])
207        else:
208            messagebox.showerror('Error', ERROR_MESSAGES['PDF_DROP_ERROR'])

210    def load_pdf(self, filename):
211        """
212            load pdf and view
213        """
214        try:
215            if self.current_pdf:
216                self.current_pdf.close()
217
218            if not os.path.exists(filename):
219                messagebox.showerror('Error', ERROR_MESSAGES['FILE_NOT_FOUND'])
220
221                return
222
223            # open file
224            self.current_pdf = fitz.open(filename)
225            self.current_page = 0
226            self.pdf_zoom = 1.0
227            self.update_zoom_display()
228
229            self.page_label['text'] = f'{self.current_page + 1} / {len(self.current_pdf)} ページ'
230            self.prev_page_button['state'] = tk.DISABLED
231            self.next_page_button['state'] = tk.NORMAL if len(self.current_pdf) > 1 else tk.DISABLED
232
233            self.display_page()
234        except PermissionError:
235            messagebox.showerror('Error', ERROR_MESSAGES['PERMISSION_DENIED'])
236        except FileNotFoundError:
237            messagebox.showerror('Error', ERROR_MESSAGES['FILE_NOT_FOUND'])
238        except Exception as e:
239            messagebox.showerror('Error', f'{ERROR_MESSAGES['UNEXPECTED_ERROR']}\n{e}')

241    def display_page(self):
242        """
243            display current page
244        """
245        if not self.current_pdf:
246            return
247
248        page = self.current_pdf[self.current_page]
249
250        zoom = self.pdf_zoom
251        mat = fitz.Matrix(zoom, zoom)
252        pix = page.get_pixmap(matrix=mat)
253
254        img = Image.frombytes('RGB', [pix.width, pix.height], pix.samples)
255        photo = ImageTk.PhotoImage(img)
256
257        self.pdf_canvas.delete('all')
258        self.pdf_canvas.config(scrollregion=(0, 0, pix.width, pix.height))
259        self.pdf_canvas.create_image(0, 0, anchor=tk.NW, image=photo)
260        self.pdf_canvas.image = photo

327    def calculate_income(self, monthly_saralies, bonus1=0, bonus2=0):
328        """
329            calculate income after employment income deduction
330            support both single monthly salary and monthly variations
331            using external configuration for income rules
332        """
333        if isinstance(monthly_saralies, int):
334            yearly_salary = (monthly_saralies * 12) + bonus1 + bonus2   # if a single monthly salary is used
335        else:
336            yearly_salary = sum(monthly_saralies) + bonus1 + bonus2     # if monthly salaries are provided
337
338        # calculate income using imported INCOME_RULES
339        for threshold, calculation in INCOME_RULES:
340            if yearly_salary <= threshold:
341                income = calculation(yearly_salary)
342
343                break
344
345        return yearly_salary, income

347    def calculate(self):
348        """
349            calculate and display results
350        """
351        try:
352            clean_input = lambda x: int(re.sub(r'[^\d]', '', x) or 0)
353
354            # get bonus values
355            bonus1 = clean_input(self.bonus1_var.get())
356            bonus2 = clean_input(self.bonus2_var.get())
357
358            # determine salary calcuation mode
359            if self.salary_mode.get() == 'single':
360                monthly = clean_input(self.monthly_var.get())
361                total, income = self.calculate_income(monthly, bonus1, bonus2)
362            else:
363                monthly_salaries = [clean_input(salary.get()) for salary in self.monthly_salaries]
364                total, income = self.calculate_income(monthly_salaries, bonus1, bonus2)
365
366            # display results
367            self.total_label['text'] = f'¥ {self.format_currency(total)}'
368            self.income_label['text'] = f'¥ {self.format_currency(income)}'
369        except ValueError:
370            # show error for invalid input
371            messagebox.showerror('Error', ERROR_MESSAGES['INVALID_INPUT'])

328    def calculate_income(self, monthly_saralies, bonus1=0, bonus2=0):
329        """
330            calculate income after employment income deduction
331            support both single monthly salary and monthly variations
332            using external configuration for income rules
333        """
334        if isinstance(monthly_saralies, int):
335            yearly_salary = (monthly_saralies * 12) + bonus1 + bonus2   # if a single monthly salary is used
336        else:
337            yearly_salary = sum(monthly_saralies) + bonus1 + bonus2     # if monthly salaries are provided
338
339        # calculate income using imported INCOME_RULES
340        for threshold, calculation in INCOME_RULES:
341            if yearly_salary <= threshold:
342                income = calculation(yearly_salary)
343
344                break
345
346        return yearly_salary, income

348    def calculate(self):
349        """
350            calculate and display results
351        """
352        try:
353            clean_input = lambda x: int(re.sub(r'[^\d]', '', x) or 0)
354
355            # get bonus values
356            bonus1 = clean_input(self.bonus1_var.get())
357            bonus2 = clean_input(self.bonus2_var.get())
358
359            # determine salary calcuation mode
360            if self.salary_mode.get() == 'single':
361                monthly = clean_input(self.monthly_var.get())
362                total, income = self.calculate_income(monthly, bonus1, bonus2)
363            else:
364                monthly_salaries = [clean_input(salary.get()) for salary in self.monthly_salaries]
365                total, income = self.calculate_income(monthly_salaries, bonus1, bonus2)
366
367            # display results
368            self.total_label['text'] = f'¥ {self.format_currency(total)}'
369            self.income_label['text'] = f'¥ {self.format_currency(income)}'
370        except ValueError:
371            # show error for invalid input
372            messagebox.showerror('Error', ERROR_MESSAGES['INVALID_INPUT'])
